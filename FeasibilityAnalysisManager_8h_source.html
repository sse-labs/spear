27struct ExpressionComperator {
28 bool operator()(const z3::expr &a, const z3::expr &b) const noexcept {
29 // Both expr must come from the same context, so we can compare their AST ids directly.
30 unsigned ia = Z3_get_ast_id(a.ctx(), a);
31 unsigned ib = Z3_get_ast_id(b.ctx(), b);
32 return ia < ib;
33 }
34};
