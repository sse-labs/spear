27struct ExpressionComperator {
28 bool operator()(const z3::expr &a, const z3::expr &b) const noexcept {
29 // Both expr must come from the same context, so we can compare their AST ids directly.
30 unsigned ia = Z3_get_ast_id(a.ctx(), a);
31 unsigned ib = Z3_get_ast_id(b.ctx(), b);
32 return ia < ib;
33 }
34};

65 struct SetKeyHash {
66 std::size_t operator()(const SetKey &key) const noexcept {
67 std::size_t hashValue = 0x9e3779b97f4a7c15ULL;
68
69 for (unsigned astId : key.astIds) {
70 hashValue ^= std::hash<unsigned>{}(astId) + 0x9e3779b97f4a7c15ULL + (hashValue << 6) + (hashValue >> 2);
71 }
72
73 return hashValue;
74 }
