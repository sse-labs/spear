25// A ModulePass that runs PhASAR's IDELinearConstantAnalysis and provides
26// a helper to query the resulting lattice values for each basic block.
27//
28// This is written for the LLVM "new" pass manager (PassBuilder).
29class PhasarHandlerPass : public llvm::PassInfoMixin<PhasarHandlerPass> {
30 public:
31 using DomainVal = psr::IDELinearConstantAnalysisDomain::l_t;
32
33 // Result: BB_name -> { var_name -> (Value*, domain_val) }
34 using BoundVarMap =
35 std::map<std::string,
36 std::map<std::string,
37 std::pair<const llvm::Value *, DomainVal>>>;
38
39 PhasarHandlerPass();
40
41 // New pass manager entry point
42 llvm::PreservedAnalyses run(llvm::Module &M,
43 llvm::ModuleAnalysisManager &AM);
44
45 // Debug helper: dump full PhASAR results
46 void dumpState() const;
47
48 void runOnModule(llvm::Module &M);
49
50 // Query all "bound variables" in a function, grouped per basic block.
51 // Requires that `run()` (and hence `runAnalysis()`) has been executed.
52 BoundVarMap queryBoundVars(llvm::Function *Func) const;
53
54 private:
55 // Backing module – only valid during/after `run()`.
56 llvm::Module *mod;
57
58 // PhASAR helper analyses
59 std::unique_ptr<psr::HelperAnalyses> HA;
60
61 // Solver results: PhASAR’s IDE solver result wrapper.
62 std::unique_ptr<psr::OwningSolverResults<
63 const llvm::Instruction *, const llvm::Value *, psr::LatticeDomain<int64_t>>>
64 AnalysisResult;
65
66 // Function entrypoints
67 std::vector<std::string> Entrypoints;
68
69 // Internal: construct analysis problem and run PhASAR solver.
70 void runAnalysis();
71};
