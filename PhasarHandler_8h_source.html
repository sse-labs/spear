114 static inline SetSatnessKey makeSetSattnessCacheEntry(
115 const Feasibility::FeasibilityAnalysisManager *Mgr,
116 const std::vector<z3::expr> &Set) {
117 // Create a cache key for the given set of formulas by hashing their AST ids in sorted order.
118 SetSatnessKey K;
119 K.Mgr = Mgr;
120 K.AstIds.reserve(Set.size());
121
122 // Inser the z3 expressions from the given set to the cache key as their AST ids, which uniquely identify
123 // the expressions. We need to sort the AST ids to ensure that the order of the formulas in the s
124 // et does not affect the hash value.
125 for (const z3::expr &E : Set) {
126 K.AstIds.push_back(Z3_get_ast_id(E.ctx(), E));
127 }
128
129 // Sort the AST ids to ensure that the order of the formulas in the set does not affect the hash value.
130 llvm::sort(K.AstIds);
131
132 // Establish uniqueness on the aST ids to ensure that the same set of formulas always results in the
133 // same cache key, even if it contains duplicate formulas.
134 K.AstIds.erase(std::unique(K.AstIds.begin(), K.AstIds.end()), K.AstIds.end());
135 return K;
136 }
